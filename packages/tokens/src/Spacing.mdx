import { Meta } from '@storybook/addon-docs/blocks';

export function SpacingBar({ name, px, cssVar }) {
  return (
    <tr>
      <td><code>{name}</code></td>
      <td><code>{cssVar}</code></td>
      <td style={{ textAlign: 'right' }}>{px}px</td>
      <td>
        <div style={{ width: px, height: 20, background: 'var(--akds-color-background-primary-default, #1E40AF)', borderRadius: 2, minWidth: 2 }} />
      </td>
    </tr>
  );
}

export function RadiusSwatch({ name, px, cssVar }) {
  return (
    <tr>
      <td><code>{name}</code></td>
      <td><code>{cssVar}</code></td>
      <td style={{ textAlign: 'right' }}>{px === 9999 ? '9999px (full)' : `${px}px`}</td>
      <td>
        <div style={{
          width: 48, height: 48,
          background: 'var(--akds-color-background-primary-default, #1E40AF)',
          borderRadius: Math.min(px, 24),
          opacity: 0.8,
        }} />
      </td>
    </tr>
  );
}

# Spacing

Spacing tokens cover the full layout system: a numeric scale for component-level spacing, layout sizes for page structure, icon and component sizing, and a border-radius scale.

---
