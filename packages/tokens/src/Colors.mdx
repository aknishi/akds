import { Meta } from '@storybook/addon-docs/blocks';
import { useDarkMode } from 'storybook-dark-mode';

export function ColorScale({ steps }) {
  return (
    <div style={{ display: 'grid', gridTemplateColumns: `repeat(${steps.length}, 1fr)`, gap: 8, marginBottom: 32 }}>
      {steps.map(([step, hex]) => (
        <div key={step} style={{ textAlign: 'center' }}>
          <div style={{ background: hex, height: 48, borderRadius: 6, border: '1px solid rgba(0,0,0,0.1)', marginBottom: 6 }} title={hex} />
          <div style={{ fontSize: 11, fontWeight: 600 }}>{step}</div>
          <div style={{ fontSize: 10, color: '#888', fontFamily: 'monospace' }}>{hex}</div>
        </div>
      ))}
    </div>
  );
}

export function LiveSwatch({ cssVar, light, dark }) {
  const isDark = useDarkMode();
  const hex = isDark ? dark : light;
  return (
    <span style={{ display: 'inline-flex', gap: 6, alignItems: 'center' }}>
      <span
        style={{
          display: 'inline-block',
          width: 20,
          height: 20,
          background: `var(${cssVar})`,
          borderRadius: 3,
          border: '1px solid rgba(127,127,127,0.2)',
          flexShrink: 0,
        }}
        title={hex}
      />
      <code style={{ fontSize: 11 }}>{hex}</code>
    </span>
  );
}

# Colors

The color system has two layers: **primitive** scales (raw color values) and **semantic** tokens (role-based, theme-aware aliases that reference primitives). Components always use semantic tokens via CSS variables.

Toggle the **dark mode** switch in the Storybook toolbar — the semantic swatches and hex values update live.

---
