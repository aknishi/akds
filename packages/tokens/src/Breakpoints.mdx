# Breakpoints

Breakpoint tokens define the responsive grid thresholds. Access them as JS values from `@aknishi/akds-tokens` or via the CSS variables.

---

## Usage

### In CSS

```css
@media (min-width: 768px) {
  /* tablet and above */
}
```

### In JavaScript

```ts
import { breakpoints } from '@aknishi/akds-tokens';
import type { Breakpoint } from '@aknishi/akds-tokens';

const tabletWidth = breakpoints.md; // 768

function useBreakpoint(bp: Breakpoint) {
  return window.matchMedia(`(min-width: ${breakpoints[bp]}px)`).matches;
}
```
